#include<stdio.h>
#include<stdlib.h>
main
{
char ca;
int ia, ib, ic, *id, ie, ig;
float fa_1, fb;
if ((ia==5)&(ib>2))  fa_1=ic+3.6*(*id);
elseif ((ia!=-5)|(ib>=2))  fa_1=ic-*id/4;
else  fa_1=(ic^2)%5;
for(ie=0;ie<=10;ie++)  /*for loop*/
for(ig=5;ig<0; ig--){
ib=(int)fa_1*2.5+(-3);
fb=(float)ie+5; fb=fa_1^3+*id;}
while(ib>5)  ib=ib-1;  //while loop
do{  fb=fb/2; ia=ia*3; *id=*id+ie;
}while(fb<1);
ig=ia&((ib|ic)|(*id))&ie;
switch(ie){
case(0)  fb=3*fa_1;
case(1)  fb=fa_1-1;
case(3)  fb=fa_1/(*id);
case(4)  fb=(*id)+fa_1;
}
scanf("%c",&ca);   /*scanf //
                     scanf*/
scanf("Number of while loop: %d %dloops",&ig, &ig);
printf("Print numbers: %d\\\t%f%d\n", ia, fb, ib);
return
}
